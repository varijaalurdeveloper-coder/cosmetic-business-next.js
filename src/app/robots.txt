    rules: [
      {
        userAgent: "*",

          "/about",
          "/products",
          "/contact",
          "/cart",

        ],
        disallow: [
          "/admin",
          "/login",

          "/checkout",
          "/order-success",
          "/*?search=*",
        ],
        crawlDelay: 1,
      },
      {
        userAgent: "Googlebot",
        allow: "/",
        disallow: ["/admin", "/login", "/orders", "/checkout"],
      },
      {
        userAgent: "Googlebot-Image",
        allow: "/",
      },
      {
        userAgent: "Bingbot",
        allow: "/",
        disallow: ["/admin", "/login"],
        crawlDelay: 1,

      {
        userAgent: "Yandex",
        allow: "/",
        disallow: ["/admin", "/login"],

      },
      {
        userAgent: "Baiduspider",
        allow: "/",
        disallow: ["/admin", "/login"],

      },
    ],
